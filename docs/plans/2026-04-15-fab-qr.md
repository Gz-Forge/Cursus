**Tech Stack:** lz-string (compresión), react-native-qrcode-svg (generar QR), expo-camera (escanear), Animated API de React Native (animación FAB).

```typescript
// src/utils/qrPayload.ts
import LZString from 'lz-string';
import { MateriaJson } from './importExport';

export const CHUNK_SIZE = 2800;

// Claves cortas para minimizar tamaño antes de comprimir
interface MateriaCompacta {
  n: number;   // numero
  s: number;   // semestre
  nm: string;  // nombre
  cd: number;  // creditos_da
  cn: number;  // creditos_necesarios
  p: string[]; // previas
  oe: number;  // oportunidades_examen
}

function aCompacta(m: MateriaJson): MateriaCompacta {
  return { n: m.numero, s: m.semestre, nm: m.nombre, cd: m.creditos_da, cn: m.creditos_necesarios, p: m.previas, oe: m.oportunidades_examen };
}

function aLarga(m: MateriaCompacta): MateriaJson {
  return { numero: m.n, semestre: m.s, nombre: m.nm, creditos_da: m.cd, creditos_necesarios: m.cn, previas: m.p, oportunidades_examen: m.oe };
}

export function encodeCarrera(materias: MateriaJson[]): string {
  const compactas = materias.map(aCompacta);
  return LZString.compressToBase64(JSON.stringify(compactas));
}

export function decodeCarrera(encoded: string): MateriaJson[] {
  const json = LZString.decompressFromBase64(encoded);
  if (!json) throw new Error('Payload QR inválido o corrupto');
  const compactas: MateriaCompacta[] = JSON.parse(json);
  return compactas.map(aLarga);
}

export interface ChunkQR {
  i: number;  // índice (0-based)
  t: number;  // total de chunks
  d: string;  // datos del chunk
}

export function splitEnChunks(encoded: string): ChunkQR[] {
  const total = Math.ceil(encoded.length / CHUNK_SIZE);
  return Array.from({ length: total }, (_, i) => ({
    i,
    t: total,
    d: encoded.slice(i * CHUNK_SIZE, (i + 1) * CHUNK_SIZE),
  }));
}

export function joinChunks(chunks: string[]): string {
  return chunks.join('');
}

export function esChunkQR(raw: string): boolean {
  try {
    const obj = JSON.parse(raw);
    return typeof obj.i === 'number' && typeof obj.t === 'number' && typeof obj.d === 'string';
  } catch {
    return false;
  }
}
```

```typescript
// src/components/FabSpeedDial.tsx
import React, { useState, useRef } from 'react';
import { View, Text, TouchableOpacity, Animated, Pressable, StyleSheet } from 'react-native';
import { useTema } from '../theme/ThemeContext';

export interface FabAction {
  icono: string;
  label: string;
  onPress: () => void;
}

interface Props {
  acciones: FabAction[];
}

export function FabSpeedDial({ acciones }: Props) {
  const [abierto, setAbierto] = useState(false);
  const tema = useTema();
  const rotacion = useRef(new Animated.Value(0)).current;
  const escalas = acciones.map(() => useRef(new Animated.Value(0)).current);
  const traducciones = acciones.map(() => useRef(new Animated.Value(0)).current);

  const abrir = () => {
    setAbierto(true);
    Animated.spring(rotacion, { toValue: 1, useNativeDriver: true }).start();
    acciones.forEach((_, i) => {
      Animated.spring(escalas[i], { toValue: 1, delay: i * 50, useNativeDriver: true }).start();
      Animated.spring(traducciones[i], { toValue: 1, delay: i * 50, useNativeDriver: true }).start();
    });
  };

  const cerrar = () => {
    setAbierto(false);
    Animated.spring(rotacion, { toValue: 0, useNativeDriver: true }).start();
    acciones.forEach((_, i) => {
      Animated.spring(escalas[i], { toValue: 0, useNativeDriver: true }).start();
      Animated.spring(traducciones[i], { toValue: 0, useNativeDriver: true }).start();
    });
  };

  const toggle = () => (abierto ? cerrar() : abrir());

  const rotar = rotacion.interpolate({ inputRange: [0, 1], outputRange: ['0deg', '45deg'] });

  return (
    <>
      {/* Overlay que cierra el menú */}
      {abierto && (
        <Pressable
          style={StyleSheet.absoluteFillObject}
          onPress={cerrar}
        />
      )}

      <View style={s.contenedor} pointerEvents="box-none">
        {/* Mini-botones (en orden inverso para que el primero quede más cerca del FAB) */}
        {[...acciones].reverse().map((accion, idx) => {
          const realIdx = acciones.length - 1 - idx;
          const offsetY = traducciones[realIdx].interpolate({
            inputRange: [0, 1],
            outputRange: [0, -((realIdx + 1) * 64)],
          });
          return (
            <Animated.View
              key={accion.label}
              style={[
                s.miniBotonContenedor,
                { transform: [{ translateY: offsetY }, { scale: escalas[realIdx] }] },
              ]}
            >
              <Text style={[s.miniLabel, { color: tema.texto, backgroundColor: tema.tarjeta }]}>
                {accion.label}
              </Text>
              <TouchableOpacity
                style={[s.miniBoton, { backgroundColor: tema.superficie, borderColor: tema.acento }]}
                onPress={() => { cerrar(); accion.onPress(); }}
              >
                <Text style={{ fontSize: 20 }}>{accion.icono}</Text>
              </TouchableOpacity>
            </Animated.View>
          );
        })}

        {/* FAB principal */}
        <TouchableOpacity
          style={[s.fab, { backgroundColor: tema.acento }]}
          onPress={toggle}
          activeOpacity={0.85}
        >
          <Animated.Text style={{ fontSize: 26, color: '#fff', transform: [{ rotate: rotar }] }}>
            +
          </Animated.Text>
        </TouchableOpacity>
      </View>
    </>
  );
}

const s = StyleSheet.create({
  contenedor: {
    position: 'absolute',
    bottom: 24,
    right: 20,
    alignItems: 'flex-end',
  },
  fab: {
    width: 56,
    height: 56,
    borderRadius: 28,
    alignItems: 'center',
    justifyContent: 'center',
    elevation: 6,
    shadowColor: '#000',
    shadowOffset: { width: 0, height: 3 },
    shadowOpacity: 0.3,
    shadowRadius: 4,
  },
  miniBotonContenedor: {
    position: 'absolute',
    bottom: 0,
    right: 0,
    flexDirection: 'row',
    alignItems: 'center',
    gap: 8,
  },
  miniBoton: {
    width: 44,
    height: 44,
    borderRadius: 22,
    alignItems: 'center',
    justifyContent: 'center',
    borderWidth: 1.5,
    elevation: 4,
  },
  miniLabel: {
    fontSize: 12,
    paddingHorizontal: 8,
    paddingVertical: 4,
    borderRadius: 6,
    overflow: 'hidden',
  },
});
```

```typescript
// src/components/QrShareModal.tsx
import React, { useMemo, useState } from 'react';
import { View, Text, Modal, TouchableOpacity, ScrollView } from 'react-native';
import QRCode from 'react-native-qrcode-svg';
import { useTema } from '../theme/ThemeContext';
import { materiasAJson } from '../utils/importExport';
import { exportarCarrera } from '../utils/importExportNative';
import { encodeCarrera, splitEnChunks, ChunkQR } from '../utils/qrPayload';
import { Materia } from '../types';

interface Props {
  visible: boolean;
  materias: Materia[];
  onCerrar: () => void;
}

export function QrShareModal({ visible, materias, onCerrar }: Props) {
  const tema = useTema();
  const [paginaActual, setPaginaActual] = useState(0);

  const chunks: ChunkQR[] = useMemo(() => {
    if (!visible || materias.length === 0) return [];
    const jsonMaterias = materiasAJson(materias);
    const encoded = encodeCarrera(jsonMaterias);
    return splitEnChunks(encoded);
  }, [visible, materias]);

  const chunkActual = chunks[paginaActual];
  const qrData = chunks.length === 1
    ? JSON.stringify({ t: 1, i: 0, d: chunkActual?.d ?? '' })
    : JSON.stringify(chunkActual);

  const irAnterior = () => setPaginaActual(p => Math.max(0, p - 1));
  const irSiguiente = () => setPaginaActual(p => Math.min(chunks.length - 1, p + 1));

  return (
    <Modal visible={visible} animationType="slide" transparent onRequestClose={onCerrar}>
      <View style={{ flex: 1, backgroundColor: 'rgba(0,0,0,0.7)', justifyContent: 'center', alignItems: 'center', padding: 20 }}>
        <View style={{ backgroundColor: tema.superficie, borderRadius: 16, padding: 24, width: '100%', alignItems: 'center' }}>

          <Text style={{ color: tema.texto, fontSize: 17, fontWeight: '700', marginBottom: 4 }}>
            Compartir carrera
          </Text>

          {chunks.length === 0 ? (
            <Text style={{ color: tema.textoSecundario, marginVertical: 20 }}>Sin materias para compartir</Text>
          ) : (
            <>
              {chunks.length > 1 && (
                <Text style={{ color: tema.acento, fontSize: 13, marginBottom: 12 }}>
                  QR {paginaActual + 1} de {chunks.length} — mostrá cada QR en orden
                </Text>
              )}

              <View style={{ backgroundColor: '#fff', padding: 12, borderRadius: 12, marginBottom: 16 }}>
                <QRCode
                  value={qrData}
                  size={220}
                  backgroundColor="#fff"
                  color="#000"
                />
              </View>

              {chunks.length > 1 && (
                <View style={{ flexDirection: 'row', gap: 16, marginBottom: 16 }}>
                  <TouchableOpacity
                    onPress={irAnterior}
                    disabled={paginaActual === 0}
                    style={{ paddingHorizontal: 20, paddingVertical: 10, backgroundColor: paginaActual === 0 ? tema.borde : tema.tarjeta, borderRadius: 8 }}
                  >
                    <Text style={{ color: paginaActual === 0 ? tema.textoSecundario : tema.texto }}>◀ Anterior</Text>
                  </TouchableOpacity>
                  <TouchableOpacity
                    onPress={irSiguiente}
                    disabled={paginaActual === chunks.length - 1}
                    style={{ paddingHorizontal: 20, paddingVertical: 10, backgroundColor: paginaActual === chunks.length - 1 ? tema.borde : tema.tarjeta, borderRadius: 8 }}
                  >
                    <Text style={{ color: paginaActual === chunks.length - 1 ? tema.textoSecundario : tema.texto }}>Siguiente ▶</Text>
                  </TouchableOpacity>
                </View>
              )}
            </>
          )}

          <TouchableOpacity
            onPress={() => exportarCarrera(materias)}
            style={{ backgroundColor: tema.tarjeta, padding: 12, borderRadius: 10, width: '100%', alignItems: 'center', marginBottom: 10 }}
          >
            <Text style={{ color: tema.texto }}>📄 Compartir como .json</Text>
          </TouchableOpacity>

          <TouchableOpacity onPress={onCerrar} style={{ padding: 10 }}>
            <Text style={{ color: tema.textoSecundario }}>Cerrar</Text>
          </TouchableOpacity>
        </View>
      </View>
    </Modal>
  );
}
```

```typescript
// src/components/QrScannerModal.tsx
import React, { useState, useEffect } from 'react';
import { View, Text, Modal, TouchableOpacity, Alert } from 'react-native';
import { CameraView, useCameraPermissions } from 'expo-camera';
import { useTema } from '../theme/ThemeContext';
import { decodeCarrera, esChunkQR, joinChunks } from '../utils/qrPayload';
import { jsonAMaterias } from '../utils/importExport';
import { useStore } from '../store/useStore';

interface Props {
  visible: boolean;
  onCerrar: () => void;
}

export function QrScannerModal({ visible, onCerrar }: Props) {
  const tema = useTema();
  const { guardarMateria } = useStore();
  const [permission, requestPermission] = useCameraPermissions();
  const [chunksRecolectados, setChunksRecolectados] = useState<string[]>([]);
  const [totalEsperado, setTotalEsperado] = useState<number | null>(null);
  const [procesando, setProcesando] = useState(false);

  // Limpiar estado al abrir
  useEffect(() => {
    if (visible) {
      setChunksRecolectados([]);
      setTotalEsperado(null);
      setProcesando(false);
    }
  }, [visible]);

  const importar = (encoded: string) => {
    try {
      const materiaJson = decodeCarrera(encoded);
      const materias = jsonAMaterias(materiaJson);
      Alert.alert(
        'Importar carrera',
        `Se encontraron ${materias.length} materias. ¿Importar?`,
        [
          { text: 'Cancelar', style: 'cancel', onPress: () => { setProcesando(false); } },
          {
            text: 'Importar',
            onPress: () => {
              materias.forEach(m => guardarMateria(m));
              onCerrar();
            },
          },
        ]
      );
    } catch {
      Alert.alert('Error', 'El QR no es válido o está dañado.');
      setProcesando(false);
    }
  };

  const onQrLeido = ({ data }: { data: string }) => {
    if (procesando) return;

    if (esChunkQR(data)) {
      const chunk = JSON.parse(data) as { i: number; t: number; d: string };

      setChunksRecolectados(prev => {
        // Evitar duplicados
        if (prev[chunk.i] !== undefined) return prev;
        const nuevos = [...prev];
        nuevos[chunk.i] = chunk.d;
        return nuevos;
      });
      setTotalEsperado(chunk.t);

      setChunksRecolectados(prev => {
        const listos = prev.filter(Boolean);
        if (listos.length === chunk.t) {
          setProcesando(true);
          const encoded = joinChunks(listos);
          setTimeout(() => importar(encoded), 300);
        }
        return prev;
      });
    } else {
      // QR simple (no chunked) — intentar decodificar directo
      setProcesando(true);
      importar(data);
    }
  };

  const chunksListos = chunksRecolectados.filter(Boolean).length;

  if (!visible) return null;

  if (!permission) {
    return (
      <Modal visible animationType="slide" transparent onRequestClose={onCerrar}>
        <View style={{ flex: 1, backgroundColor: 'rgba(0,0,0,0.8)', justifyContent: 'center', alignItems: 'center' }}>
          <Text style={{ color: '#fff' }}>Cargando permisos...</Text>
        </View>
      </Modal>
    );
  }

  if (!permission.granted) {
    return (
      <Modal visible animationType="slide" transparent onRequestClose={onCerrar}>
        <View style={{ flex: 1, backgroundColor: 'rgba(0,0,0,0.8)', justifyContent: 'center', alignItems: 'center', padding: 32 }}>
          <Text style={{ color: '#fff', fontSize: 16, textAlign: 'center', marginBottom: 20 }}>
            Se necesita permiso de cámara para escanear QR
          </Text>
          <TouchableOpacity onPress={requestPermission}
            style={{ backgroundColor: tema.acento, padding: 14, borderRadius: 10, marginBottom: 12 }}>
            <Text style={{ color: '#fff', fontWeight: '700' }}>Dar permiso</Text>
          </TouchableOpacity>
          <TouchableOpacity onPress={onCerrar}>
            <Text style={{ color: tema.textoSecundario }}>Cancelar</Text>
          </TouchableOpacity>
        </View>
      </Modal>
    );
  }

  return (
    <Modal visible animationType="slide" onRequestClose={onCerrar}>
      <View style={{ flex: 1, backgroundColor: '#000' }}>
        <CameraView
          style={{ flex: 1 }}
          facing="back"
          barcodeScannerSettings={{ barcodeTypes: ['qr'] }}
          onBarcodeScanned={procesando ? undefined : onQrLeido}
        />

        {/* Overlay de progreso multi-QR */}
        <View style={{ position: 'absolute', top: 0, left: 0, right: 0, padding: 20, paddingTop: 60, alignItems: 'center' }}>
          {totalEsperado !== null && totalEsperado > 1 ? (
            <View style={{ backgroundColor: 'rgba(0,0,0,0.7)', borderRadius: 12, padding: 12 }}>
              <Text style={{ color: '#fff', fontSize: 16, textAlign: 'center' }}>
                Escaneado {chunksListos} de {totalEsperado} QRs
              </Text>
              <View style={{ flexDirection: 'row', gap: 6, marginTop: 8, justifyContent: 'center' }}>
                {Array.from({ length: totalEsperado }).map((_, i) => (
                  <View
                    key={i}
                    style={{
                      width: 16, height: 16, borderRadius: 8,
                      backgroundColor: chunksRecolectados[i] ? tema.acento : 'rgba(255,255,255,0.3)',
                    }}
                  />
                ))}
              </View>
            </View>
          ) : (
            <View style={{ backgroundColor: 'rgba(0,0,0,0.7)', borderRadius: 12, padding: 12 }}>
              <Text style={{ color: '#fff', fontSize: 14 }}>Apuntá la cámara al QR</Text>
            </View>
          )}
        </View>

        {/* Botón cerrar */}
        <View style={{ position: 'absolute', bottom: 40, left: 0, right: 0, alignItems: 'center' }}>
          <TouchableOpacity
            onPress={onCerrar}
            style={{ backgroundColor: 'rgba(0,0,0,0.7)', paddingHorizontal: 32, paddingVertical: 14, borderRadius: 30 }}
          >
            <Text style={{ color: '#fff', fontSize: 16 }}>Cancelar</Text>
          </TouchableOpacity>
        </View>
      </View>
    </Modal>
  );
}
```
